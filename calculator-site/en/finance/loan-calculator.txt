          navLinks.classList.remove('active');
          hamburger.classList.remove('active');
        }
      });

      langSelect.addEventListener('change', function() {
        const selectedLang = this.value;
        let targetUrl = '';
        
        // Get current path and replace language segment
        const currentPath = window.location.pathname;
        const pathSegments = currentPath.split('/');

        // Determine the base path for the finance section
        let financePath = '';
        if (currentPath.includes('/finance/')) {
          const financeIndex = pathSegments.findIndex(segment => segment === 'finance');
          if (financeIndex !== -1) {
            financePath = pathSegments.slice(financeIndex).join('/').replace('finance/loan-calculator', 'finance/loan-calculator');
          }
        }

        // Build new URL based on selected language
        switch(selectedLang) {
          case 'zh':
            targetUrl = financePath ? `/zh/finance/${financePath}` : '/zh/finance/loan-calculator.html';
            break;
          case 'es':
            targetUrl = financePath ? `/es/finance/${financePath}` : '/es/finance/loan-calculator.html';
            break;
          case 'de':
            targetUrl = financePath ? `/de/finance/${financePath}` : '/de/finance/loan-calculator.html';
            break;
          case 'fr':
            targetUrl = financePath ? `/fr/finance/${financePath}` : '/fr/finance/loan-calculator.html';
            break;
          default: // English
            targetUrl = financePath ? `/en/finance/${financePath}` : '/en/finance/loan-calculator.html';
        }

      // Set current language based on URL
      const pathSegments = window.location.pathname.split('/');

      // Loan Calculator Functionality
      const loanAmountInput = document.getElementById('loan-amount');
      const interestRateInput = document.getElementById('interest-rate');
      const loanTermInput = document.getElementById('loan-term');
      const paymentPerYearInput = document.getElementById('payment-per-year');
      const calculateBtn = document.getElementById('calculate-btn');
      const resultContainer = document.getElementById('result-container');
      const paymentAmount = document.getElementById('payment-amount');
      const totalInterest = document.getElementById('total-interest');
      const loanSummary = document.getElementById('loan-summary');
      const amortizationTable = document.getElementById('amortization-table');

      // Calculate Loan function
      function calculateLoan() {
        const loanAmount = parseFloat(loanAmountInput.value);
        const interestRate = parseFloat(interestRateInput.value) / 100;
        const loanTerm = parseInt(loanTermInput.value);
        const paymentsPerYear = parseInt(paymentPerYearInput.value);

        if (!loanAmount || !interestRate || !loanTerm || loanAmount <= 0 || interestRate <= 0 || loanTerm <= 0) {
          alert('Please enter valid loan amount, interest rate, and loan term.');
          return;
        }

        // Calculate monthly interest rate
        const paymentFrequency = paymentsPerYear;
        const periodInterestRate = interestRate / paymentFrequency;
        const totalPayments = loanTerm * paymentFrequency;

        // Calculate payment amount using amortization formula
        const numerator = periodInterestRate * Math.pow(1 + periodInterestRate, totalPayments);
        const denominator = Math.pow(1 + periodInterestRate, totalPayments) - 1;
        let paymentAmountValue = loanAmount * (numerator / denominator);

        // Round to nearest cent
        paymentAmountValue = Math.round(paymentAmountValue * 100) / 100;

        // Calculate total interest paid
        const totalPaymentsAmount = paymentAmountValue * totalPayments;
        const totalInterestPaid = totalPaymentsAmount - loanAmount;

        // Update payment amount and total interest
        paymentAmount.textContent = paymentAmountValue.toFixed(2);
        totalInterest.textContent = `Total Interest: $${totalInterestPaid.toFixed(0)}`;

        // Update loan summary
        const paymentFrequencyText = paymentPerYearInput.options[paymentPerYearInput.selectedIndex].text.split(' ')[0];
        const summaryHTML = `
          <h3><i class="fas fa-money-bill-wave"></i> Loan Summary</h3>
          <p>Your estimated ${paymentFrequencyText.toLowerCase()} payment is $${paymentAmountValue.toFixed(2)} based on a $${loanAmount.toFixed(0)} loan with ${interestRate * 100}% interest over ${loanTerm} years.</p>
          <h3><i class="fas fa-chart-line"></i> Total Cost</h3>
          <p>The total cost of the loan will be $${totalPaymentsAmount.toFixed(0)}, which includes $${totalInterestPaid.toFixed(0)} in interest payments.</p>
        `;

        loanSummary.innerHTML = summaryHTML;

        // Generate amortization schedule
        const tableBody = amortizationTable.querySelector('tbody');
        tableBody.innerHTML = '';

        let remainingBalance = loanAmount;
        let paymentNumber = 1;
        let currentPeriod = new Date();
        currentPeriod.setMonth(currentPeriod.getMonth() + 1); // Start next month

        for (let i = 0; i < totalPayments; i++) {
          const interestPayment = remainingBalance * periodInterestRate;
          const principalPayment = paymentAmountValue - interestPayment;
          remainingBalance -= principalPayment;

          const row = document.createElement('tr');

          // Format payment date
          const paymentDate = new Date(currentPeriod);
          const monthNames = ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun', 
                             'Jul', 'Aug', 'Sep', 'Oct', 'Nov', 'Dec'];
          const dateStr = `${monthNames[paymentDate.getMonth()]} ${paymentDate.getFullYear()}`;

          row.innerHTML = `
            <td>${paymentNumber}</td>
            <td>${dateStr}</td>
            <td>$${principalPayment.toFixed(2)}</td>
            <td>$${interestPayment.toFixed(2)}</td>
            <td>$${remainingBalance.toFixed(2)}</td>
            <td>$${paymentAmountValue.toFixed(2)}</td>
          `;

          tableBody.appendChild(row);

          // Increment date
          currentPeriod.setMonth(currentPeriod.getMonth() + 1);
          paymentNumber++;
        }

        // Show result container
        resultContainer.style.display = 'block';
      }

      // Add event listener to calculate button
      calculateBtn.addEventListener('click', calculateLoan);

      // Also allow pressing Enter in input fields
      loanAmountInput.addEventListener('keypress', (e) => {
        if (e.key === 'Enter') calculateLoan();
      });

      interestRateInput.addEventListener('keypress', (e) => {
        if (e.key === 'Enter') calculateLoan();
      });

      loanTermInput.addEventListener('keypress', (e) => {
        if (e.key === 'Enter') calculateLoan();
      });

      paymentPerYearInput.addEventListener('keypress', (e) => {
        if (e.key === 'Enter') calculateLoan();
      });
    })();
