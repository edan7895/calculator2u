  // Get available domains
  async function fetchDomains() {
    try {
      const res = await fetch(apiBase + '/domains');
      const data = await res.json();
      domains = data['hydra:member'].map(d => d.domain);
    } catch {
      domains = ['mail.tm']; // Fallback
    }
  }

  // Register account (random username + current domain)
  async function registerAccount(domain) {
    const username = 'user' + Math.random().toString(36).substring(2, 10);
    const address = `${username}@${domain}`;
    const password = 'Pass1234!';

    // Register
    try {
      await fetch(apiBase + '/accounts', {
        method: 'POST',
        headers: {'Content-Type':'application/json'},
        body: JSON.stringify({address, password})
      });
    } catch (e) {
      // Account might exist, try to login anyway
    }
    return {address, password};
  }

  // Login to get token
  async function login(address, password) {
    const res = await fetch(apiBase + '/token', {
      method: 'POST',
      headers: {'Content-Type':'application/json'},
      body: JSON.stringify({address, password})
    });
    if(!res.ok) throw new Error('Login failed');
    const data = await res.json();
    return data.token;
  }

  // Initialize email account (register + login)
  async function initEmailAccount() {
    notify(t('loading'));
    if(domains.length === 0) await fetchDomains();
    const domain = domains[currentDomainIndex];
    account = await registerAccount(domain);
    token = await login(account.address, account.password);
    emailBox.textContent = account.address;
    codeBox.textContent = t('noMail');
    lastMailId = null;
    if(checkMailInterval) clearInterval(checkMailInterval);
    checkMailInterval = setInterval(checkEmails, 8000);
  }

  // Get email list
  async function fetchEmails() {
    if(!token) return [];
    try {
      const res = await fetch(apiBase + '/messages', {
        headers: {Authorization: `Bearer ${token}`}
      });
      if(!res.ok) return [];
      const data = await res.json();
      return data['hydra:member'] || [];
    } catch {
      return [];
    }
  }

  // Get email content
  async function fetchEmailContent(id) {
    if(!token) return null;
    try {
      const res = await fetch(apiBase + `/messages/${id}`, {
        headers: {Authorization: `Bearer ${token}`}
      });
      if(!res.ok) return null;
      return await res.json();
    } catch {
      return null;
    }
  }

  // Extract verification code (4-8 digit numbers)
  function extractCode(text) {
    const match = text.match(/\b(\d{4,8})\b/);
    return match ? match[1] : null;
  }

  // Check for new emails
  async function checkEmails() {
    const mails = await fetchEmails();
    if(mails.length === 0) {
      codeBox.textContent = t('noMail');
      return;
    }
    if(mails[0].id !== lastMailId) {
      lastMailId = mails[0].id;
      const mail = await fetchEmailContent(lastMailId);
      if(mail) {
        let code = extractCode(mail.text) || extractCode(mail.html) || t('noCode');
        codeBox.textContent = code;
        notify(t('newMail'));
        notifySound.play();
      }
    }
  }

  // Clear mailbox (refresh display)
  async function clearMailbox() {
    notify(t('clearing'));
    codeBox.textContent = t('noMail');
    lastMailId = null;
    setTimeout(() => notify(t('cleared')), 1000);
  }

  // Switch to new email
  async function switchNewEmail() {
    notify(t('switchingEmail'));
    currentDomainIndex = 0;
    await initEmailAccount();
    notify(t('switchedEmail'));
  }

  // Switch domain
  async function switchDomain() {
    notify(t('switchingDomain'));
    currentDomainIndex = (currentDomainIndex + 1) % domains.length;
    await initEmailAccount();
    notify(t('switchedDomain'));
  }
