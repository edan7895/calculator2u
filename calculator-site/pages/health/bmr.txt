let currentLang = 'en';
let useMetric = true;

function switchLang() {
  currentLang = currentLang === 'en' ? 'zh' : 'en';
  langSwitchBtn.textContent = currentLang === 'en' ? '中文' : 'EN';
  
  document.querySelectorAll('[data-i18n]').forEach(el => {
    const key = el.getAttribute('data-i18n');
    if (i18nText[currentLang][key]) {
      if (el.tagName === 'INPUT') {
        el.placeholder = i18nText[currentLang][key];
      } else {
        el.textContent = i18nText[currentLang][key];
      }
    }
  });

  if (useMetric) {
    unitSwitchBtn.textContent = i18nText[currentLang]['unit'];
  } else {
    unitSwitchBtn.textContent = i18nText[currentLang]['unit-alt'];
  }
}

function switchUnit() {
  useMetric = !useMetric;
  form.weight.placeholder = useMetric ? i18nText[currentLang]['label-weight'] : 'Weight (lb)';
  form.height.placeholder = useMetric ? i18nText[currentLang]['label-height'] : 'Height (in)';
  form.weight.value = '';
  form.height.value = '';
  unitSwitchBtn.textContent = useMetric 
    ? i18nText[currentLang]['unit'] 
    : i18nText[currentLang]['unit-alt'];
}

function calculateBMR(gender, age, weight, height) {
  if (useMetric) {
    if (gender === "male") {
      return 10 * weight + 6.25 * height - 5 * age + 5;
    } else {
      return 10 * weight + 6.25 * height - 5 * age - 161;
    }
  } else {
    weight *= 0.45359237; // lb to kg
    height *= 2.54; // in to cm
    if (gender === "male") {
      return 10 * weight + 6.25 * height - 5 * age + 5;
    } else {
      return 10 * weight + 6.25 * height - 5 * age - 161;
    }
  }
}

function updateChart(bmr) {
  const ctx = document.getElementById('bmr-chart').getContext('2d');
  const data = {
    labels: ['BMR', 'Daily Needs', 'Light Activity', 'Moderate Activity', 'Heavy Activity'],
    datasets: [{
      label: 'Calories',
      data: [bmr, bmr * 1.2, bmr * 1.375, bmr * 1.55, bmr * 1.725],
      backgroundColor: [
        'rgba(75, 192, 192, 0.6)',
        'rgba(54, 162, 235, 0.6)',
        'rgba(255, 206, 86, 0.6)',
        'rgba(255, 159, 64, 0.6)',
        'rgba(255, 99, 132, 0.6)'
      ],
      borderColor: [
        'rgba(75, 192, 192, 1)',
        'rgba(54, 162, 235, 1)',
        'rgba(255, 206, 86, 1)',
        'rgba(255, 159, 64, 1)',
        'rgba(255, 99, 132, 1)'
      ],
      borderWidth: 1
    }]
  };

  if (!window.bmrChart) {
    window.bmrChart = new Chart(ctx, {
      type: 'bar',
      data: data,
      options: {
        responsive: true,
        scales: {
          y: {
            beginAtZero: true,
            title: {
              display: true,
              text: currentLang === 'en' ? 'Calories per day' : '每日卡路里'
            }
          },
          x: {
            title: {
              display: true,
              text: currentLang === 'en' ? 'Activity Level' : '活动水平'
            }
          }
        }
      }
    });
  } else {
    window.bmrChart.data = data;
    window.bmrChart.options.scales.y.title.text = currentLang === 'en' ? 'Calories per day' : '每日卡路里';
    window.bmrChart.options.scales.x.title.text = currentLang === 'en' ? 'Activity Level' : '活动水平';
    window.bmrChart.update();
  }
}

function initialize() {
  switchLang();
}

langSwitchBtn.addEventListener('click', switchLang);
unitSwitchBtn.addEventListener('click', switchUnit);

form.addEventListener('submit', e => {
  e.preventDefault();
  const gender = document.getElementById('gender').value;
  const age = parseInt(document.getElementById('age').value);
  const weight = parseFloat(document.getElementById('weight').value);
  const height = parseFloat(document.getElementById('height').value);
  
  if (isNaN(age) || isNaN(weight) || isNaN(height)) return;
  
  const bmr = calculateBMR(gender, age, weight, height);
  bmrValueEl.textContent = Math.round(bmr);
  
  bmrAdviceEl.textContent = gender === 'male' 
    ? i18nText[currentLang]['advice-male'] 
    : i18nText[currentLang]['advice-female'];
  
  updateChart(bmr);
  
  document.getElementById('result-container').scrollIntoView({
    behavior: 'smooth'
  });
});

window.addEventListener('load', initialize);
